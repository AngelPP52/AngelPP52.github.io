                      极简 Antlr4 实践 -- 基于 Golang 的 MySQL 生成器

                      shougang.deng 2020-12-20 CompilerTheory,Golang

                      # 业务背景

                      1. 项目接入了多个国家，每个国家需要单独提 SQL 文件
                      2. 团队人员增多，与 QA 制定了Checklist 中的 SQL 提单规范
                      3. 由于历史问题，逻辑 DB 的名称不规范
                      4. 我们目前一共有 3 个国家，每个国家 1-3 个逻辑DB，同时主要的业务有 1000 个分表，一次大的需求会有几万行 SQL 需要执行，review 成本和修改成本都比较高。

                      # 技术目标

                      # 实现团队 SQL 规范

                      1. 同一个表的所有 column 变更合并为一句 SQL

                      2. 每一个逻辑 DB 汇总一个 SQL 文件

                      3. 表名和列名不得有 MySQL 保留字

                      # 词法分析

                      词法分析，实际上就是分词，比如汉语中，“我喜欢打篮球”，就可以分词为“我”、“喜欢”、“打”、“篮球”。分词就是把连续的字符串转化为一个个程序可以识别的基本单位（称为 token）。

                      我们本次的工作主要针对建表 SQL 及修改表的 SQL 进行分词，相对于完整的 SQL 语法简单很多。

                      # 语法分析

                      词法是把一个句子识别为一个个 token 串，语法分析则是组装这些 token，生成实际的语法结构，最后输出为 AST（抽象语法树）。

                      比如“2+3*5"，输出的 AST 如下所示：

                      # 语义分析

                      语义分析是根据具体的语法，来消除 AST 中的语法结构的歧义。比如在作用域 1 和作用域 2 中都有变量 a，程序应该使用a1 还是 a2，亦或是抛出异常？

                      这也是各种编程语言的静态检查中常用的技术。

                      # 结果生成

                      拿到 AST 树及树中各节点对应的语义之后，我们就可以生成结果代码了。

                      # Antlr 简介

                      Java 编写的基于LL(*)算法实现的语法解析器生成器，名称取自 ANother Tool for Language Recognition。

                      简单来说，Antlr 是一个可以开发编译器的编译器，生成代码生成器的代码生成器。

                      我们会用它来完成词法分析和语法分析。

                      # Antlr 优点

                      1. 多语言支持，支持 C、C++、Java、Go、Python 等多种编程语言
                      2. 语法相对简单
                      3. 有大量已经定义好的语法规则，不必自己手写规则了。

                      # 为什么使用编译技术而不是正则等方法实现 MySQL Generator?

                      1. MySQL 是一门完整的编程语言，语法复杂
                      2. 正则维护困难，难以拓展需求，同时无法处理复杂的业务场景
                      3. 代码生成技术是一门非常有用的技术，学会使用它受益匪浅，可以大幅提高程序员的工作幸福感。

                      # 代码解析

                      Antlr 很轻松的就帮我们完成了词法分析和语法分析的部分，当然这也和本示例非常简单有关系，我们接下来要完成的就是最终的业务代码部分了。

                      要编写业务代码，最核心的其实是理解 antlr 的 visitor 模式。visitor 模式，就是每当 Antlr 访问到一个 expression 时，都会告诉我们，然后我们决定做些什么处理。

                      我们先看一下 Antlr生成的 calcparser_visitor.go 文件，核心是 CalcParserVisitor 接口，包含有以下4 个方法：

   // Visit a parse tree produced by CalcParser#Operation.
	 VisitOperation(ctx *OperationContext) interface{}
}

                      type CalcVisitor struct {
   parser.BaseCalcParserVisitor
   r int
}

func (v *CalcVisitor) Result() int {
   return v.r
}

// 遍历节点
func (v *CalcVisitor) visitRule(node antlr.RuleNode) interface{} {
   return node.Accept(v)
}

func (v *CalcVisitor) VisitStart(ctx *parser.StartContext) interface{} {
   return v.visitRule(ctx.Expression())
}

func (v *CalcVisitor) VisitNumber(ctx *parser.NumberContext) interface{} {
   i, err := strconv.Atoi(ctx.NUMBER().GetText())
   if err != nil {
      panic(err)
   }
   return i
}

// 核心计算逻辑
func (v *CalcVisitor) VisitOperation(ctx *parser.OperationContext) interface{} {
   left := v.visitRule(ctx.Expression(0))
   right := v.visitRule(ctx.Expression(1))
   t := ctx.GetOp()
   switch t.GetTokenType() {
   case parser.CalcParserMUL:
      v.r = left.(int) * right.(int)
      return v.r
   case parser.CalcParserDIV:
      if right.(int) == 0 {
         panic("divide by zero")
      }
      v.r = left.(int) * right.(int)
      return v.r
   case parser.CalcParserADD:
      v.r = left.(int) + right.(int)
      return v.r
   case parser.CalcParserSUB:
      v.r = left.(int) - right.(int)
      return v.r
   }
   panic("unsupported expression")
}

   input := "1+2*3"

func calcByVisitor(input string) int {
   // Setup the input
   is := antlr.NewInputStream(input)

   // Create the Lexer
   lexer := parser.NewCalcLexer(is)
   stream := antlr.NewCommonTokenStream(lexer, antlr.TokenDefaultChannel)

   // Create the Parser
   p := parser.NewCalcParser(stream)

   var v = &visitor.CalcVisitor{}
   // Start is rule name of CalcLexer.g4
   p.Start().Accept(v)
   return v.Result()
}

                      # 2.构建数据

                      这一步，就是把 Antlr 的数据转化为我们的自定义数据。

                      这里我是一个一个对照着定义好的语法树来进行开发的，相关的代码在 /visitor 目录下面。

                      核心思路是每个子 handler 将数据返回，然后交给父 handler 进行组装，不断向父级递归获得最终期望的数据结构。

                      Create Table 解析:

                      // Visit a parse tree produced by MySqlParser#queryCreateTable.
func (v *Visitor) VisitQueryCreateTable(ctx *parser.QueryCreateTableContext) interface{} {
   tableName := ctx.TableName().GetText()
   v.cTable = &model.Table{
      TableName: tableName,
   }
   val := v.visit(ctx.GetRuleContext())
   v.CreateTables = append(v.CreateTables, v.cTable)
   v.cTable = nil
   return val
}

// Visit a parse tree produced by MySqlParser#columnCreateTable.
func (v *Visitor) VisitColumnCreateTable(ctx *parser.ColumnCreateTableContext) interface{} {
   tableName := ctx.TableName().GetText()
   var options []string
   for _, option := range ctx.AllTableOption() {
      t := getFullText(option)
      t = strings.TrimSpace(t)
      if t != "" {
         options = append(options, t)
      }
   }
   v.cTable = &model.Table{
      TableName:    tableName,
      TableOptions: strings.Join(options, ","),
   }
   val := v.visit(ctx.CreateDefinitions())
   v.CreateTables = append(v.CreateTables, v.cTable)
   v.cTable = nil
   return val
}

                      Alter Table 解析:

                      func (v *Visitor) VisitAlterTable(ctx *parser.AlterTableContext) interface{} {
   for _, spec := range ctx.AllAlterSpecification() {
      tableName := getFullText(ctx.TableName())
      var alterSQLs *model.AlterSQL
      if val, ok := v.AlterSQLMap[tableName]; ok {
         alterSQLs = val
      } else {
         alterSQLs = &model.AlterSQL{
            TableName: getFullText(ctx.TableName()),
         }
         v.AlterSQLMap[tableName] = alterSQLs
      }
      res := v.visit(spec.GetRuleContext())
      var alterSQL = new(model.AlterSQLUnit)
      switch spec.GetRuleContext().(type) {
      case *parser.AlterByAddColumnContext:
         alterSQL.Column = res.(*model.Column)
         alterSQL.Operation = model.AlterTableAddColumn
      case *parser.AlterByModifyColumnContext, *parser.AlterByChangeColumnContext:
         alterSQL.Column = res.(*model.Column)
         alterSQL.Operation = model.AlterTableModifyColumn
      case *parser.AlterByAddIndexContext:
         alterSQL.Index = res.(*model.Index)
         alterSQL.Operation = model.AlterTableAddIndex
      case *parser.AlterByDropIndexContext:
         alterSQL.Index = res.(*model.Index)
         alterSQL.Operation = model.AlterTableDropIndex
      }
      alterSQLs.AlterSQLUnits = append(alterSQLs.AlterSQLUnits, alterSQL)
   }
   return nil
}

func (v *Visitor) VisitAlterByAddColumn(ctx *parser.AlterByAddColumnContext) interface{} {
   uid := getFullText(ctx.Uid(0))
   column := v.visit(ctx.ColumnDefinition()).(*model.Column)
   column.Name = uid
   return column
}

func (v *Visitor) VisitAlterByChangeColumn(ctx *parser.AlterByChangeColumnContext) interface{} {
   uid := getFullText(ctx.Uid(1))
   column := v.visit(ctx.ColumnDefinition()).(*model.Column)
   column.Name = uid
   return column
}

func (v *Visitor) VisitAlterByModifyColumn(ctx *parser.AlterByModifyColumnContext) interface{} {
   uid := getFullText(ctx.Uid(0))
   column := v.visit(ctx.ColumnDefinition()).(*model.Column)
   column.Name = uid
   return column
}

func (v *Visitor) VisitAlterByAddUniqueKey(ctx *parser.AlterByAddUniqueKeyContext) interface{} {
   indexName := getFullText(ctx.GetIndexName())

      Name:        indexName,

                      在上一步中，我们已经从 input 组装好了期望的数据，接下来，就是将数据再按照预期的格式生成。有了数据之后，对数据的操作都非常简单。

                      type SQLGenerator struct {
   CreateTables         []*model.Table
   CreateTablesMap      map[string]*model.Table
   AlterSQLList         []*model.AlterSQL
   TableNameAlterSQLMap map[string]*model.AlterSQL

   UpdateStatements []*model.UpdateStatement

   IsTest          bool
   UnionAll        bool
   UnionSum        bool
   CheckColumnName bool
}

func NewGenerator() *SQLGenerator {
   return &SQLGenerator{
      CreateTablesMap:      map[string]*model.Table{},
      TableNameAlterSQLMap: map[string]*model.AlterSQL{},
   }
}

func (g *SQLGenerator) GenSQLs() (result []string) {
   for _, table := range g.CreateTables {
      if g.CheckColumnName {
         err := checkColumnsName(table.Columns...)
         if err != nil {
            panic(err)
         }
      }
      sqlList := table.GenSQLsWithPartition()
      result = append(result, sqlList...)
   }
   for _, alterSQLs := range g.AlterSQLList {
      sqlList := alterSQLs.GenSQLsWithPartition()
      result = append(result, sqlList...)
   }
   for _, statement := range g.UpdateStatements {
      sqlList := statement.GenSQLsWithPartition()
      result = append(result, sqlList...)
   }
   return result
}

                      # 尾声

                      代码生成技术对于开发人员来说还是有很高价值的，特别是针对那些重复的编码工作。 Golang 目前还不支持泛型，所以就有开发人员使用代码生成技术，通过遍历类型以生成兼容多种数据类型的数据结构。 服务的 grpc 迁移工作，使用代码生成技术可以通过抽取服务端的 func 定义，一键完成相关元数据及中间代码的生成。 比如异步任务，我们也可以只定义 task handler，然后通过代码生成技术来生成其他相关联的代码，减少开发和维护成本。
